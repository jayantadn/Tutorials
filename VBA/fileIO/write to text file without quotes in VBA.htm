    Try the 'print' command instead of the 'write' command. Also, it's best to close your file.

    Sub Print_Test()
    Dim record As String
    Open "C:myfile.txt" For Output As #1
    record = "Message1"
    Print #1, record
    record = "Message2"
    Print #1, record
    Close #1
    End Sub
